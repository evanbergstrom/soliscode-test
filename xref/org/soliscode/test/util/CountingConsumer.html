5   /// Counts how many times the [accept()][Consumer#accept] method is called.
6   ///
7   /// @param <T> The type of the parameter for the `accept` method.
8   /// @author evanbergstrom
9   /// @since 1.0
10  public class CountingConsumer<T> implements Consumer<T> {
11  
12      private int count = 0;
13  
14      /// Counts each time this method is called.
15      /// @param t This parameter is ignored.
16      @Override
17      public void accept(final T t) {
18          count++;
19      }
20  
21      /// Returns the number of times the `accept` method has been called.
22      /// @return the number of times the `accept` method has been called.
23      public int count() {
24          return count;
25      }
26  }
