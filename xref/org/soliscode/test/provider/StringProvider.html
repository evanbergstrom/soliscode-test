17  package org.soliscode.test.provider;
18  
19  import org.jetbrains.annotations.NotNull;
20  
21  import java.util.Objects;
22  
23  /// Provides instances of an [String] for the purposes of testing.
24  ///
25  /// @author evanbergstrom
26  /// @since 1.0
27  /// @see String
28  public class StringProvider implements ObjectProvider<String> {
29  
30      @Override
31      public @NotNull String defaultInstance() {
32          return "";
33      }
34  
35      /// Creates an instance of the string class with a value equal to the string representation of a primitive integer
36      /// value.
37      ///
38      /// @param seed a primitive integer value.
39      /// @return an instance of the `String` class
40      @Override
41      public @NotNull String createInstance(int seed) {
42          return String.valueOf(seed);
43      }
44  
45      /// Creates a copy of the string value. Since instances of [String] are immutable, the copy is not performed and
46      /// the value provided as an argument is returned.
47      /// @param other the string value to copy.
48      /// @return the string value.
49      /// @throws NullPointerException if the argument is `null`.
50      @Override
51      public @NotNull String copyInstance(final @NotNull String other) {
52          return Objects.requireNonNull(other);
53      }
54  }
