17  package org.soliscode.test.provider;
18  
19  import org.jetbrains.annotations.NotNull;
20  
21  import java.util.*;
22  import java.util.function.Supplier;
23  import java.util.stream.Stream;
24  
25  /// Provides [Object] instances for use in testing.
26  ///
27  /// @implSpec
28  /// Implementations of this interface must provide two methods:
29  /// - [createInstance(int)][#createInstance(int)]
30  /// - [copyInstance()][#copyInstance(T)]
31  ///
32  /// @param <T> the type being tested.
33  /// @author evanbergstrom
34  /// @since 1.0.0
35  public interface ObjectProvider<T> {
36  
37      /// Creates a single instance of class being tested using the default constructor.
38      /// @return  an instance of class.
39      @NotNull T defaultInstance();
40  
41      /// Creates a single instance of class being tested based upon an integer seed value.
42      ///
43      /// @implSpec
44      /// Calls to this method with different seed values should return objects with different values.
45      ///
46      /// @param   seed an integer seed to use to create and instance of the class.
47      /// @return  an instance of class.
48      @NotNull T createInstance(int seed);
49  
50      /// Copies an instance of the class being tested. The copy will satisfy equality semantics with the original
51      /// instance being copied.
52      ///
53      /// @implSpec
54      /// The copy of the object should satisfy the equality semantics with the original object (_i.e._ for `o1` and
55      /// `o2`, `o1.equals(o2)` should return `true` and `o1.hashCode() == o2.hashCode()` should evaluate to `true`.
56      ///
57      /// @param      other the instance to copy.
58      /// @return     a copy of the instance.
59      /// @throws     NullPointerException if the argument is a null value.
60      /// @complexity _constant time_.
61      @NotNull T copyInstance(@NotNull T other);
62  
63      /// Creates a single instance of class being tested.
64      ///
65      /// @implSpec
66      /// Each call must return a valid object instance, but there is no requirement that successive calls to the method
67      /// return instances that satisfy equality semantics, or that they be unique. The default implementation is the
68      /// same as calling `createInstance(0)`.
69      ///
70      /// @return     an instance of class.
71      /// @complexity _constant time_.
72      default @NotNull T createInstance(){
73          return createInstance(0);
74      }
75  
76      /// Returns the number of unique instances of the class that can be created.
77      ///
78      /// @implSpec
79      /// The default implementation returns [Integer#MAX_VALUE]. This should be overridden for enumerations or any classes
80      /// that have a bounded number of unique values.
81      ///
82      /// @return     the maximum number of unique instances that can be created.
83      /// @complexity _constant time_.
84      @SuppressWarnings("SameReturnValue")
85      default int uniqueSizeLimit() {
86          return Integer.MAX_VALUE;
87      }
88  
89      /// Creates a [Supplier] that will generate equal instances of the class being provided.
90      ///
91      /// @implSpec
92      /// The supplier should produce equal instances of the class for each the call to `get()`.
93      ///
94      /// @implNote
95      /// The default implementation will call `createInstance(0)` to create each instance.
96      ///
97      /// @return     A supplier the returns instances of the class that are equal.
98      /// @complexity _constant time_.
99      default @NotNull Supplier<T> equalInstanceSupplier() {
100         final ObjectProvider<T> provider = this;
101         return new Supplier<>() {
102             private final T orig = provider.createInstance(0);
103 
104             @Override
105             public T get() {
106                 return provider.copyInstance(orig);
107             }
108         };
109     }
110 
111     /// Creates a [Supplier] that will generate unique instances of the class being provided. Uniqueness will be
112     /// guaranteed for all instances created by this supplier. Instances may not be unique from multiple suppliers.
113     ///
114     /// @implSpec
115      /// The supplier should produce unique instances of the class for each the call to `get()`, and it should be callable
116      /// a number of times up to the value returned by `uniqueSizeLimit()`. If it is called more that `uniqueSizeLimit`
117      /// times, it should throw an [IllegalStateException].
118      ///
119      /// @implNote
120      /// The default implementation will call `createInstance(int)` with successive integers starting with 0.
121      ///
122      /// @return A supplier the returns instances of the class that are unique.
123      /// @throws IllegalStateException if `get` is called on the supplier more than `uniqueSizeLimit` times.
124      /// @complexity _constant time_.
125     default @NotNull Supplier<T> uniqueInstanceSupplier() {
126         return uniqueInstanceSupplier(0);
127     }
128 
129     /// Creates a [Supplier] that will generate unique instances of the class being provided. Uniqueness will be
130     /// guaranteed for all instances created by this supplier. Instances may not be unique from multiple suppliers.
131     ///
132     /// @implSpec
133     /// The supplier should produce unique instances of the class for each the call to `get()`, and it should be callable
134     /// a number of times up to the value returned by `uniqueSizeLimit()`. If it is called more that `uniqueSizeLimit`
135     /// times, it should throw an [IllegalStateException].
136     ///
137     /// @implNote
138     /// The default implementation will call `createInstance(int)` with successive integers starting with 0.
139     ///
140     /// param seed the initial seed to use to crate the first instance.
141     /// @param seed the seed to use to create elements.
142     /// @return A supplier the returns instances of the class that are unique.
143     /// @throws IllegalStateException if `get` is called on the supplier more than `uniqueSizeLimit` times.
144     /// @complexity _constant time_.
145     default @NotNull Supplier<T> uniqueInstanceSupplier(final int seed) {
146         final ObjectProvider<T> provider = this;
147         return new Supplier<>() {
148             private int i = seed;
149 
150             @Override
151             public T get() {
152                 if (i >= uniqueSizeLimit()) {
153                     throw new IllegalStateException("cannot create " + i + " unique instances, limit is " + uniqueSizeLimit());
154                 }
155                 return provider.createInstance(i++);
156             }
157         };
158     }
159 
160     /// Creates a [Supplier] that will generate random instances of the class being provided. This should be used where
161     /// the individual values of the instances is not important, such here we are testing for performance of some part
162     /// of the implementation, but where creating instances with successive seed values might bias the algorithm (_e.g._
163     /// hash code distribution). In most cases, the methods {@link #uniqueInstanceSupplier()} should be
164     /// preferred so that the test data is stable across executions.
165     ///
166     /// @implSpec
167     /// The supplier should produce random instances of the class for each the call to `get()`. The order of the objects
168     /// created should be stable across instances of the supplier.
169     ///
170     /// @implNote
171     /// The default implementation will call `createInstance(int)` with random integers between 0 and the value returned
172     /// by `upperUniqueLimit()`.
173     ///
174     /// @return A supplier the returns random instances of the class.
175     default @NotNull Supplier<T> randomInstanceSupplier() {
176         final ObjectProvider<T> provider = this;
177         return new Supplier<>() {
178             private final Random r = new Random();
179 
180             @Override
181             public T get() {
182                 return provider.createInstance(r.nextInt(uniqueSizeLimit()));
183             }
184         };
185     }
186 
187     /// Creates s list of instances that are equal to each other.
188     ///
189     /// @param size The number of equal instances to create.
190     /// @return A list of equal instances.
191     /// @complexity _linear time_ based upon the number of instance (the `size` argument).
192     default @NotNull List<T> createEqualObjects(int size) {
193         return Stream.generate(equalInstanceSupplier()).limit(size).toList();
194     }
195 
196     /// Creates multiple instances of class being tested with unique values.
197     ///
198     /// @param size the number of instances to create.
199      /// @return a list of the created instances.
200      /// @throws IllegalArgumentException if size is greater than `uniqueSizeLimit()`
201      /// @complexity _linear time_ based upon the number of instance (the `size` argument).
202     default @NotNull List<T> createUniqueInstances(int size) {
203         return createUniqueInstances(size, 0);
204     }
205 
206     /// Creates multiple instances of class being tested with unique values.
207     ///
208     /// @param size the number of instances to create.
209     /// @param seed the seed to use to create elements.
210     /// @return a list of the created instances.
211     /// @throws IllegalArgumentException if size is greater than `uniqueSizeLimit()`
212     /// @complexity _linear time_ based upon the number of instance (the `size` argument).
213     default @NotNull List<T> createUniqueInstances(int size, int seed) {
214         if (size > uniqueSizeLimit()) {
215             throw new IllegalArgumentException("cannot create " + size + " unique instances, limit is " + uniqueSizeLimit());
216         }
217         return new ArrayList<>(Stream.generate(uniqueInstanceSupplier(seed)).limit(size).toList());
218     }
219 
220     /// Creates multiple instances of class being tested with random values. This should be used where the individual
221     /// values of the instances is not important, such as here we are testing for performance of some part of the
222     /// implementation (_e.g._ hash code distribution). In most cases, the method {@link #createUniqueInstances(int)}
223     /// should be preferred so that the test data is stable across executions.
224     ///
225     /// @param size the number of instances to create.
226     /// @return a list of the created instances.
227     /// @complexity _linear time_ based upon the number of instance (the `size` argument).
228     default @NotNull List<T> createRandoInstances(int size) {
229         return new ArrayList<>(Stream.generate(randomInstanceSupplier()).limit(size).toList());
230     }
231 }
