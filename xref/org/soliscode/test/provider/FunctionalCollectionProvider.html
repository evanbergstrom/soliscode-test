3   import org.jetbrains.annotations.NotNull;
4   
5   import java.util.*;
6   import java.util.function.Function;
7   import java.util.function.Supplier;
8   
9   /// A collection provider that uses a set of functions to construct instances of the collection. There are four
10  /// elements required to create a collection provider using this class:
11  /// - Default constructor function
12  /// - Copy constructor function
13  /// - Collection constructor function
14  /// - Element provider.
15  /// For example, to create a collection provider for an `ArrayList` of Integer objects:
16  /// ```java
17  ///     CollectionProvider<Integer> provider = FunctionalCollectionProvider<Integer, ArrayList>(
18  ///         ArrayList;:new, ArrayList::new, ArrayList::new, Providers.integerProvider());
19  /// ```
20  ///
21  /// @param <E> the element type for the iterable
22  /// @param <I> the iterable type.
23  /// @author evanbergstrom
24  /// @since 1.0.0
25  public class FunctionalCollectionProvider<E, I extends Iterable<E>> extends FunctionalProvider<I>
26          implements CollectionProvider<E, I> {
27  
28      /// Default number of elements to use for a test.
29      private static final int DEFAULT_SIZE = 10;
30  
31      private final @NotNull Function<Collection<E>, I> collectionConstructor;
32      private final @NotNull ObjectProvider<E> elementProvider;
33  
34      /// Create an instance of this collection provider that uses the methods and element provider specified in the
35      /// arguments for its implementation.
36      /// @param defaultConstructor the supplier to use to create default instances of the collection.
37      /// @param copyConstructor the function to use to create a copy of the collection.
38      /// @param collectionConstructor the function to use to create an instance from another collection.
39      /// @param elementProvider the element provider.
40      /// @throws NullPointerException if any of the arguments are `null`
41      public FunctionalCollectionProvider(final @NotNull Supplier<I> defaultConstructor,
42                                          final @NotNull Function<I, I> copyConstructor,
43                                          final @NotNull Function<Collection<E>, I> collectionConstructor,
44                                          final @NotNull ObjectProvider<E> elementProvider) {
45          super(defaultConstructor, copyConstructor);
46          this.collectionConstructor = Objects.requireNonNull(collectionConstructor);
47          this.elementProvider = Objects.requireNonNull(elementProvider);
48      }
49  
50      @Override
51      public @NotNull ObjectProvider<E> elementProvider() {
52          return elementProvider;
53      }
54  
55      /// Creates an instance of iterable with no elements.
56      /// @return and instance of iterable
57      @Override
58      public @NotNull I emptyInstance() {
59          return defaultInstance();
60      }
61  
62      @Override
63      public @NotNull I createInstance(final @NotNull Collection<E> c) {
64          return collectionConstructor.apply(c);
65      }
66  
67      @Override
68      public @NotNull I createInstance(int seed) {
69          return createInstance(elementProvider().createUniqueInstances(DEFAULT_SIZE, seed));
70      }
71  
72      @Override
73      public @NotNull I createSingleton() {
74          return createInstance(Collections.singleton(elementProvider.createInstance()));
75      }
76  
77      /// Creates an instance of iterable with a single element.
78      /// @param e the element for the iterable.
79      /// @return and instance of iterable.
80      @Override
81      public @NotNull I createSingleton(E e) {
82          return createInstance(Collections.singleton(e));
83      }
84  
85      /// Creates an instance of iterable with the same elements provided.
86      /// @param elements the elements for the iterable.
87      /// @return and instance of iterable.
88      /// @throws NullPointerException if the argument is `null`.
89      @Override
90      public  @NotNull I createInstance(@NotNull E[] elements) {
91          return createInstance(Arrays.asList(elements));
92      }
93  
94      @Override
95      public @NotNull I createInstanceWithUniqueElements() {
96          return createInstance(elementProvider.createUniqueInstances(DEFAULT_SIZE));
97      }
98  
99      @Override
100     public @NotNull I createInstanceWithUniqueElements(int size) {
101         return createInstance(elementProvider.createUniqueInstances(size));
102     }
103 
104     @Override
105     public @NotNull I createInstanceWithUniqueElements(int size, int seed) {
106         return createInstance(elementProvider.createUniqueInstances(size, seed));
107     }
108 
109     /// Create an instance of this collection provider that uses the methods and element provider specified in the
110     /// arguments for its implementation.
111     /// @param <E> the type of the elements.
112     /// @param <I> the type of the iterable.
113     /// @param defaultConstructor the supplier to use to create default instances of the collection.
114     /// @param copyConstructor the function to use to create a copy of the collection.
115     /// @param collectionConstructor the function to use to create an instance from another collection.
116     /// @param elementProvider the element provider.
117     /// @return the collection provider.
118     /// @throws NullPointerException if any of the arguments are `null`
119     public static <E, I extends Iterable<E>> FunctionalCollectionProvider<E, I> from(
120             final @NotNull Supplier<I> defaultConstructor,
121             final @NotNull Function<I, I> copyConstructor,
122             final @NotNull Function<Collection<E>, I> collectionConstructor,
123             final @NotNull ObjectProvider<E> elementProvider) {
124         return new FunctionalCollectionProvider<>(defaultConstructor, copyConstructor, collectionConstructor, elementProvider);
125     }
126 }
