8   /// Test suite for classes that implement the [List] interface. When implementing  this class, the only method that will
9   ///  need to be implemented is [org.soliscode.test.contract.support.CollectionContractSupport#provider()].
10  ///
11  /// @param <E> The type of the elements.
12  /// @param <L> The type of the list.
13  /// @author evanbergstrom
14  /// @since 1.0
15  public interface ListContract<E, L extends List<E>> extends SequencedCollectionContract<E, L>,
16          ListAddContract<E, L>,
17          ListToArrayContract<E, L>,
18          GetContract<E, L>,
19          SetContract<E, L>,
20          IndexOfContract<E, L>,
21          LastIndexOfContract<E, L>,
22          AddAtPositionContract<E, L>,
23          RemoveAtPositionContract<E, L>,
24          ReplaceAllContract<E, L>,
25          SortContract<E, L>,
26          SubListContract<E, L> {
27  
28      default void doesNotSupportModification() {
29          SequencedCollectionContract.super.doesNotSupportModification();
30          doesNotSupportMethod(CollectionMethods.Set);
31          doesNotSupportMethod(CollectionMethods.AddAtIndex);
32          doesNotSupportMethod(CollectionMethods.RemoveAtIndex);
33          doesNotSupportMethod(CollectionMethods.Sort);
34          doesNotSupportMethod(CollectionMethods.RetainAll);
35      }
36  }
