13  /// This interface tests if a list class has implemented the [sort][List#add(Object)] method correctly.
14  ///
15  /// @param <E> The element type being tested.
16  /// @param <L> The list type being tested.
17  /// @author evanbergstrom
18  /// @since 1.0
19  /// @see List#add(Object)
20  public interface ListAddContract<E, L extends List<E>> extends AddContract<E, L> {
21  
22      /// Tests that the [add][List#add] method works.
23      @DisplayName("Test that the add method appends to the end of the list")
24      @Test
25      default void testAdd() {
26          if (supportsMethod(CollectionMethods.Add)) {
27              List<E> list = provider().emptyInstance();
28              List<E> values = elementProvider().createUniqueInstances(DEFAULT_SIZE);
29              for (int i = 0; i < values.size(); i++) {
30                  E element = values.get(i);
31                  list.add(element);
32                  assertEquals(element, list.get(i));
33              }
34          } else {
35              List<E> list = provider().emptyInstance();
36              assertThrows(UnsupportedOperationException.class, () -> list.add(elementProvider().createInstance()));
37          }
38      }
39  
40  }
