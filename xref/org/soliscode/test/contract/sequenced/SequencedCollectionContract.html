10  ///
11  /// @param <E> The element type being tested.s
12  /// @param <C> The collection type being tested.

14  /// @see SequencedCollection
15  /// @since 1.0
16  public interface SequencedCollectionContract<E, C extends SequencedCollection<E>>
17          extends CollectionContract<E, C>,
18          AddFirstContract<E, C>,
19          AddLastContract<E, C>,
20          GetFirstContract<E, C>,
21          GetLastContract<E, C>,
22          RemoveFirstContract<E, C>,
23          RemoveLastContract<E, C>,
24          ReversedContract<E, C> {

26  
27      /// This method can be called when testing collection classes that do not permit modification.
28      default void doesNotSupportModification() {
29          CollectionContract.super.doesNotSupportModification();
30          doesNotSupportMethod(CollectionMethods.AddFirst);
31          doesNotSupportMethod(CollectionMethods.AddLast);
32          doesNotSupportMethod(CollectionMethods.RemoveFirst);
33          doesNotSupportMethod(CollectionMethods.RemoveLast);
34      }
