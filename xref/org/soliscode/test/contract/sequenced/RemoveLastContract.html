13  /// Test for the `removeLast method` in the [SequencedCollection] interface. This contract class can be used individually
14  /// by a test class, but it is normally used through the [SequencedCollectionContract] class:
15  /// ```java
16  /// public class MyCollectionTest extends SequencedCollectionContract<Integer, MyCollection<Integer>> {
17  /// }
18  /// ```
19  /// If a test is using the SequencedCollectionContract class, but the class being tested does not implement the `removeLast`
20  /// method based upon the specification in the `SequencedCollection` class, then it can be omitted from the tests using the
21  /// `doesNotSupportMethod()` method:
22  /// ```java
23  /// public class MyCollectionTest extends SequencedCollectionContract<Integer, MyCollection<Integer>> {
24  ///     public MyCollectionTest() {
25  ///         doesNotSupportMethod(CollectionMethods.RemoveLast);
26  ///     }
27  /// }
28  /// ```
29  /// @param <E> The element type being tested.
30  /// @param <C> The collection type being tested.
31  /// @author evanbergstrom
32  /// @since 1.0
33  public interface RemoveLastContract<E, C extends SequencedCollection<E>> extends CollectionContractSupport<E, C> {
34  
35      /// Tests that the [removeLast][SequencedCollection#removeLast] method works with a collection with elements.
36      ///
37      /// @throws org.opentest4j.AssertionFailedError if the test fails.
38      /// @see SequencedCollection#removeLast
39      @DisplayName("Test that the removeLast method works")
40      @Test
41      default void testRemoveLast() {
42          List<E> elements = elementProvider().createUniqueInstances(DEFAULT_SIZE);
43          if (supportsMethod(CollectionMethods.RemoveLast)) {
44              SequencedCollection<E> collection = provider().emptyInstance();
45              collection.addAll(elements);
46  
47              for (int i=elements.size()-1; i >= 0; i--) {
48                  E removed = collection.removeLast();
49                  assertEquals(elements.get(i), removed);
50                  assertFalse(collection.contains(removed));
51              }
52          } else {
53              SequencedCollection<E> collection = provider().createInstance(elements);
54              assertThrows(UnsupportedOperationException.class, collection::removeLast);
55          }
56      }
57  
58      /// Tests that the [removeLast][SequencedCollection#removeLast] method works for an empty collection.
59      ///
60      /// @throws org.opentest4j.AssertionFailedError if the test fails.
61      /// @see SequencedCollection#removeLast
62      @DisplayName("Test that the removeLast method throws for an empty collection")
63      @Test
64      default void testRemoveLastOnEmptyCollection() {
65          if (supportsMethod(CollectionMethods.RemoveLast)) {
66              SequencedCollection<E> collection = provider().emptyInstance();
67              assertThrows(NoSuchElementException.class, collection::removeLast);
68          } else {
69              SequencedCollection<E> collection = provider().emptyInstance();
70              assertThrowsAny(List.of(UnsupportedOperationException.class, NoSuchElementException.class),
71                      collection::removeLast);
72          }
73      }
74  }
