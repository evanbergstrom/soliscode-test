13  /// This interface tests if a class has implemented the `addLast()` method correctly based upon the specification in the
14  /// [SequencedCollection] class. This contract class can be used individually by a test class, but it is normally used
15  /// through the [SequencedCollectionContract] class:
16  /// ```java
17  /// public class MyCollectionTest extends SequencedCollectionContract<Integer, MyCollection<Integer>> {
18  /// }
19  /// ```
20  /// If a test is using the SequencedCollectionContract class, but the class being tested does not implement the `addLast`
21  /// method based upon the specification in the `SequencedCollection` class, then it can be omitted from the tests using the
22  /// `doesNotSupportMethod()` method:
23  /// ```java
24  /// public class MyCollectionTest extends SequencedCollectionContract<Integer, MyCollection<Integer>> {
25  ///     public MyCollectionTest() {
26  ///         doesNotSupportMethod(CollectionMethods.AddLast);
27  ///     }
28  /// }
29  /// ```
30  /// @param <E> The element type being tested.
31  /// @param <C> The collection type being tested.
32  /// @author evanbergstrom
33  /// @see SequencedCollection#addLast
34  /// @since 1.0
35  public interface AddLastContract<E, C extends SequencedCollection<E>> extends CollectionContractSupport<E, C> {
36  
37      /// Tests that the [addLast][SequencedCollection#addLast] method adds elements to the end of the collection.
38      ///
39      /// @throws org.opentest4j.AssertionFailedError if the test fails.
40      /// @see SequencedCollection#addLast
41      @DisplayName("Test that the addLast method adds elements to the end of the collection")
42      @Test
43      default void testAddLast() {
44          SequencedCollection<E> collection = provider().emptyInstance();
45          if (supportsMethod(CollectionMethods.AddLast)) {
46              List<E> values = elementProvider().createUniqueInstances(DEFAULT_SIZE);
47              for (int i = 0; i < values.size(); i++) {
48                  E element = values.get(i);
49                  collection.addLast(element);
50                  assertTrue(collection.contains(element));
51                  assertEquals(i + 1, collection.size());
52                  assertEquals(collection.getLast(), element);
53              }
54          } else {
55              assertThrows(UnsupportedOperationException.class, () -> collection.addLast(elementProvider().createInstance()));
56          }
57      }
58  
59      /// Tests that the [addFirst][SequencedCollection#addFirst] method handles null values correctly.
60      ///
61      /// @throws org.opentest4j.AssertionFailedError if the test fails.
62      /// @see SequencedCollection#addLast
63      @DisplayName("Test that the addFirst method works with null element values")
64      @Test
65      default void testAddLastWithNullValue() {
66          SequencedCollection<E> collection = provider().emptyInstance();
67          if (supportsMethod(CollectionMethods.AddLast)) {
68              if (permitNulls()) {
69                  collection.addLast(null);
70                  assertTrue(collection.contains(null));
71                  assertEquals(1, collection.size());
72              } else {
73                  assertThrows(NullPointerException.class, () -> collection.addLast(null));
74              }
75          }
76      }
77  
78      /// Tests that the [addFirst][SequencedCollection#addFirst] method handles duplicate values correctly.
79      ///
80      /// @throws org.opentest4j.AssertionFailedError if the test fails.
81      /// @see SequencedCollection#addLast
82      @DisplayName("Test that the addFirst method works with duplicate element values")
83      @Test
84      default void addLastWithDuplicateValue() {
85          if (supportsMethod(CollectionMethods.AddLast)) {
86              if (permitDuplicates()) {
87                  List<E> values = elementProvider().createUniqueInstances(DEFAULT_SIZE);
88                  SequencedCollection<E> collection = provider().createInstance(values);
89                  for (int i = 0; i < values.size(); i++) {
90                      collection.addLast(values.get(i));
91                      assertTrue(collection.contains(values.get(i)));
92                      assertEquals(values.size() + i + 1, collection.size());
93                  }
94              } else {
95                  SequencedCollection<E> collection = provider().emptyInstance();
96                  E value = elementProvider().createInstance();
97                  collection.addLast(value);
98                  E otherValue = elementProvider().copyInstance(value);
99                  collection.addLast(otherValue);
100                 assertEquals(1, collection.size());
101             }
102         }
103     }
