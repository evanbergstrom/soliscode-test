13  /// Test for the getFirst method in the [SequencedCollection] interface. This contract class can be used individually
14  /// by a test class, but it is normally used through the [SequencedCollectionContract] class:
15  /// ```java
16  /// public class MyCollectionTest extends SequencedCollectionContract<Integer, MyCollection<Integer>> {
17  /// }
18  /// ```
19  /// If a test is using the SequencedCollectionContract class, but the class being tested does not implement the `getLast`
20  /// method based upon the specification in the `SequencedCollection` class, then it can be omitted from the tests using the
21  /// `doesNotSupportMethod()` method:
22  /// ```java
23  /// public class MyCollectionTest extends SequencedCollectionContract<Integer, MyCollection<Integer>> {
24  ///     public MyCollectionTest() {
25  ///         doesNotSupportMethod(CollectionMethods.GetLast);
26  ///     }
27  /// }
28  /// ```
29  /// @param <E> The element type being tested.
30  /// @param <C> The collection type being tested.
31  /// @author evanbergstrom
32  /// @since 1.0
33  public interface GetLastContract<E, C extends SequencedCollection<E>> extends CollectionContractSupport<E, C> {
34  
35      /// Tests that the [getLast][SequencedCollection#getLast] method works.
36      ///
37      /// @throws org.opentest4j.AssertionFailedError if the test fails.
38      /// @see SequencedCollection#getLast
39      @DisplayName("Test that the getLast method works")
40      @Test
41      default void testGetLast() {
42          if (supportsMethod(CollectionMethods.GetLast)) {
43              List<E> elements = elementProvider().createUniqueInstances(2);
44              SequencedCollection<E> collection = provider().createInstance(elements);
45              assertEquals(elements.get(1), collection.getLast());
46          }
47      }
48  
49      /// Tests that the [getLast][SequencedCollection#getLast] method works on an empty collection.
50      ///
51      /// @throws org.opentest4j.AssertionFailedError if the test fails.
52      /// @see SequencedCollection#getLast
53      @DisplayName("Test that the getLast method throws for an empty collection")
54      @Test
55      default void testGetLastOnEmptyCollection() {
56          if (supportsMethod(CollectionMethods.GetLast)) {
57              SequencedCollection<E> collection = provider().emptyInstance();
58              assertThrows(NoSuchElementException.class, collection::getLast);
59          }
60      }
61  }
