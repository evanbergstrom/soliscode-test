9   /// Test for the reversed method in the [SequencedCollection] interface. This contract class can be used individually
10  /// by a test class, but it is normally used through the [SequencedCollectionContract] class:
11  /// ```java
12  /// public class MyCollectionTest extends SequencedCollectionContract<Integer, MyCollection<Integer>> {
13  /// }
14  /// ```
15  /// If a test is using the SequencedCollectionContract class, but the class being tested does not implement the `reversed`
16  /// method based upon the specification in the `SequencedCollection` class, then it can be omitted from the tests using the
17  /// `doesNotSupportMethod()` method:
18  /// ```java
19  /// public class MyCollectionTest extends SequencedCollectionContract<Integer, MyCollection<Integer>> {
20  ///     public MyCollectionTest() {
21  ///         doesNotSupportMethod(CollectionMethods.Reversed);
22  ///     }
23  /// }
24  /// ```
25  /// @param <E> The element type being tested.
26  /// @param <C> The element type being tested.
27  /// @author evanbergstrom
28  /// @since 1.0
29  public interface ReversedContract<E, C extends SequencedCollection<E>> extends CollectionContractSupport<E, C> {
30  
31      /// Tests that the [reversed][SequencedCollection#reversed] method returns a view =of the collection with the
32      /// elements in the reverse ordering.
33      ///
34      /// @throws org.opentest4j.AssertionFailedError if the test fails.
35      /// @see SequencedCollection#reversed
36      @Test
37      @DisplayName("the reverse methods returns a reversed view of the collection.")
38      default void testReversed() {
39          if (supportsMethod(CollectionMethods.Reversed)) {
40              List<E> values = elementProvider().createUniqueInstances(DEFAULT_SIZE);
41              SequencedCollection<E> collection = provider().createInstance(values);
42  
43              SequencedCollection<E> reversed = collection.reversed();
44              Iterator<E> iterator = reversed.iterator();
45              for (int i = values.size() - 1; i >= 0; i--) {
46                  Assertions.assertEquals(values.get(i), iterator.next());
47              }
48          }
49      }
50  }
