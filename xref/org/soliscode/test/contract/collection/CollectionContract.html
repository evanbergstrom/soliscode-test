12  /// Test suite for classes that implement the {@link Collection} interface. When implementing  this class, the only
13  /// method that will need to be implemented is [CollectionContractSupport#provider()]. Also, the following methods
14  /// will allow the tests to be configured based upon the desired behavior of the collection class being tested:
15  ///
16  /// - [permitNulls][CollectionContractConfig#permitNulls]: Specifies if the collection permits null value elements.
17  /// - [permitDuplicates][CollectionContractConfig#permitDuplicates]: Specifies if the collection permits duplicate values.
18  /// - [permitIncompatibleTypes][CollectionContractConfig#permitIncompatibleTypes]: Specifies if the collection allows the
19  ///       search for incompatible values.
20  ///
21  /// For example, most of the collections from the JDK permit null values, permit duplicate values, and permit the
22  /// search for incompatible types. To create a test class for ArrayList, the following constructor would be
23  /// appropriate:
24  /// ```java
25  ///     public class ArrayListTest
26  ///         extends AbstractTest<Integer>
27  ///         implements TestCollection<ArrayList<Integer>, Integer>, WithIntegerElement<ArrayList<Integer>> {
28  ///
29  ///     public ArrayListTest() {
30  ///         permitNulls(true);
31  ///         permitIncompatibleTypes(true);
32  ///         permitDuplicates(true);
33  ///     }
34  /// }
35  /// ```
36  ///
37  /// @param <E> The element type being tested.s
38  /// @param <C> The collection type being tested.
39  /// @author evanbergstrom
40  /// @see Collection
41  /// @since 1.0
42  public interface CollectionContract<E, C extends Collection<E>> extends IterableContract<E, C>,
43          CollectionContractConfig,
44          AddContract<E,C>,
45          AddAllContract<E,C>,
46          ClearContract<E, C>,
47          ContainsContract<E, C>,
48          ContainsAllContract<E, C>,
49          IsEmptyContract<E, C>,
50          RemoveContract<E, C>,
51          RemoveAllContract<E, C>,
52          RemoveIfContract<E, C>,
53          RetainAllContract<E, C>,
54          SizeContract<E, C>,
55          StreamContract<E, C>,
56          ToArrayContract<E, C> {
57  
58      @Override
59      boolean supportsMethod(final OptionalMethod method);
60  
61      /// Used to indicate that the class being tested does not support an optional method.
62      /// @param method the method that the class being tested does not support.
63      void doesNotSupportMethod(final OptionalMethod method);
64  
65      /// Specific if the test collection supports the methods that allow modification. It is a convenience function to
66      /// set the support state for all the modification methods at once. These methods are:
67      ///
68      /// - [Collection#add(Object)]
69      /// - [Collection#addAll(Collection)]
70      /// - [Collection#clear]
71      /// - [Collection#remove(Object)]
72      /// - [Collection#removeAll(Collection)]
73      /// - [Collection#removeIf(java.util.function.Predicate)]
74      /// - [Collection#retainAll(Collection)]
75      /// - [Iterator#remove()]
76      default void doesNotSupportModification() {
77          doesNotSupportMethod(CollectionMethods.Add);
78          doesNotSupportMethod(CollectionMethods.AddAll);
79          doesNotSupportMethod(CollectionMethods.Clear);
80          doesNotSupportMethod(CollectionMethods.Remove);
81          doesNotSupportMethod(CollectionMethods.RemoveAll);
82          doesNotSupportMethod(CollectionMethods.RemoveIf);
83          doesNotSupportMethod(CollectionMethods.RetainAll);
84          doesNotSupportMethod(CollectionMethods.IteratorRemove);
85      }
86  }
