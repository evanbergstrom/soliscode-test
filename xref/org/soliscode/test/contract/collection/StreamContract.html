14  /// This interface tests if a collection class has implemented the `stream()` method
15  /// correctly and that the iterator that is returned satisfies the [Stream] contract.
16  ///
17  /// @param <E> The element type being tested.
18  /// @param <C> The collection type being tested.
19  /// @author evanbergstrom
20  /// @see Collection#stream
21  /// @see Stream
22  /// @since 1.0
23  public interface StreamContract<E, C extends Collection<E>> extends CollectionContractSupport<E, C> {
24  
25      /// Tests that the `stream` method works for an empty collection.
26      @Test
27      @DisplayName("The stream method can be called on an empty collection")
28      default void testStreamForEmptyCollection() {
29          Collection<E> collection = provider().emptyInstance();
30          Stream<E> stream = collection.stream();
31          assertEquals(0, stream.count());
32      }
33  
34      /// Tests that the `stream` method works for a collection with elements.
35      @Test
36      @DisplayName("The stream method can be called on a collection with elements")
37      default void testStreamForCollectionWithElements() {
38          Collection<E> collection = provider().createInstanceWithUniqueElements();
39          Collection<E> elements = collection.stream().toList();
40          assertSameSize(collection, elements);
41          assertEqualsByIdentity(elements, collection);
42      }
43  
44      /// Tests that the `parallelStream` method works for an empty collection.
45      @Test
46      @DisplayName("The parallelStream method can be called on an empty collection")
47      default void testParallelStreamForEmptyCollection() {
48          Collection<E> collection = provider().emptyInstance();
49          Stream<E> stream = collection.parallelStream();
50          assertEquals(0, stream.count());
51      }
52  
53      /// Tests that the `parallelStream` method works for a collection with elements.
54      @Test
55      @DisplayName("The parallelStream method can be called on a collection with elements")
56      default void testParallelStreamForCollectionWithElements() {
57          Collection<E> collection = provider().createInstanceWithUniqueElements();
58          Collection<E> elements = collection.parallelStream().toList();
59          assertSameSize(collection, elements);
60          assertEqualsByIdentity(elements, collection);
61      }
62  }
