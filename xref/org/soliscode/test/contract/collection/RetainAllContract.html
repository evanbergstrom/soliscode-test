15  /// This interface tests if a collection class has implemented the `retainAll` method correctly.
16  ///
17  /// @param <E> The element type being tested.
18  /// @param <C> The collection type being tested.
19  /// @author evanbergstrom
20  /// @see Collection#retainAll
21  /// @since 1.0.0
22  public interface RetainAllContract<E, C extends Collection<E>> extends CollectionContractSupport<E, C> {
23  
24      /// Tests that the `retainAll` method works on an empty container.
25      @Test
26      @DisplayName("The retainAll method can be called on an empty container")
27      default void testRetainAllOnEmptyContainer() {
28          Collection<E> collection = provider().emptyInstance();
29          Collection<E> values = elementProvider().createUniqueInstances(2);
30          if (supportsMethod(CollectionMethods.RetainAll)) {
31              boolean changed = collection.retainAll(values);
32              assertFalse(changed);
33          } else {
34              assertThrows(UnsupportedOperationException.class, () -> collection.removeIf((e) -> true));
35          }
36      }
37  
38      /// Test that the `retainAll` method works on a container with elements.
39      @Test
40      @DisplayName("The retainAll method works on a container with elements")
41      default void testRetainAllOnContainerWithElements() {
42          List<E> values = elementProvider().createUniqueInstances(DEFAULT_SIZE);
43          Collection<E> collection = provider().createInstance(values);
44          if (supportsMethod(CollectionMethods.RemoveAll)) {
45              // Remove the remaining elements
46              boolean changed = collection.retainAll(values.subList(1, values.size() - 1));
47              assertTrue(changed);
48              assertEquals(values.size() - 2, collection.size());
49              assertContainsAll(collection, values);
50              assertFalse(collection.contains(values.getFirst()));
51              assertFalse(collection.contains(values.getLast()));
52          } else {
53              assertThrows(UnsupportedOperationException.class, () -> collection.remove(values.getFirst()));
54          }
55      }
56  
57      /// Tests that the `retainAll` method works with incompatible objects.
58      @Test
59      @DisplayName("The retainAll method works with incompatible types")
60      default void testRetainAllOnIncompatibleObject() {
61          if (supportsMethod(CollectionMethods.RemoveAll)) {
62              Collection<E> collection = provider().createInstanceWithUniqueElements();
63              assertTrue(collection.retainAll(Collections.singleton(new MatchNothing())));
64              assertTrue(collection.isEmpty());
65          }
66      }
67  
68      /// Tests that the `retainAll` method throws on a null collection.
69      /// # Implementation Notes
70      /// This test checks that the `retainAll` function throws the correct exception if it is called on a null
71      /// value. IntelliJ will detect a problem when it is called with a null argument since the method declaration
72      /// has a NotNull annotation. Since this is what we are trying to test, the inspection is suppressed here.
73      ///
74      /// Any implementations that use the `NotNull` annotation for the collection parameter may throw an
75      /// `IllegalArgumentException` here, so either exception type is accepted.
76      @Test
77      @DisplayName("The retainAll method throws on a null collection")
78      @SuppressWarnings("DataFlowIssue")
79      default void testRetainAllThrowsOnNullCollection() {
80          if (supportsMethod(CollectionMethods.RetainAll)) {
81              Collection<E> collection = provider().emptyInstance();
82              Assertions.assertThrowsAny(List.of(NullPointerException.class, IllegalArgumentException.class),
83                      () -> collection.retainAll(null));
84          }
85      }
86  }
