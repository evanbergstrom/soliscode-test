7   import java.util.Collection;
8   
9   import static org.junit.jupiter.api.Assertions.assertFalse;
10  import static org.junit.jupiter.api.Assertions.assertTrue;
11  
12  /// This interface tests if a collection class has implemented the `isEmpty` method
13  /// correctly.
14  ///
15  /// @param <E> The element type being tested.
16  /// @param <C> The collection type being tested.
17  /// @author evanbergstrom
18  /// @see Collection#isEmpty
19  /// @since 1.0
20  public interface IsEmptyContract<E, C extends Collection<E>> extends CollectionContractSupport<E, C> {
21  
22      /// Tests that the `isEmpty` method returns `true` for an empty collection.
23      @Test
24      @DisplayName("Test isEmpty returns true for an empty collection")
25      default void testIsEmptyForEmptyCollection() {
26          Collection<E> collection = provider().emptyInstance();
27          assertTrue(collection.isEmpty());
28      }
29  
30      /// Tests that the `isEmpty` method returns `false` for a collection with
31      /// at least one element.
32      @Test
33      @DisplayName("Test isEmpty returns false for a collection with elements")
34      default void testIsEmptyForNonEmptyCollection() {
35          Collection<E> collection = provider().createSingleton();
36          assertFalse(collection.isEmpty());
37      }
38  }
