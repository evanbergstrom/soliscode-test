16  /// This interface tests if a collection class has implemented the `removeAll` method correctly.
17  ///
18  /// @param <E> The element type being tested.
19  /// @param <C> The collection type being tested.
20  /// @author evanbergstrom
21  /// @see Collection#removeAll
22  /// @since 1.0
23  public interface RemoveAllContract<E, C extends Collection<E>> extends CollectionContractSupport<E, C> {
24  
25      /// Test that the `removeAll` method works on an empty container.
26      @Test
27      @DisplayName("The removeAll method can be called on an empty container")
28      default void testRemoveAllOnEmptyContainer() {
29          Collection<E> collection = provider().emptyInstance();
30          List<E> values = elementProvider().createUniqueInstances(DEFAULT_SIZE);
31          if (supportsMethod(CollectionMethods.RemoveAll)) {
32              boolean changed = collection.removeAll(values);
33              assertFalse(changed);
34          } else {
35              assertThrows(UnsupportedOperationException.class, () -> collection.removeAll(values));
36          }
37      }
38  
39      /// Test that the `removeAll` method works on a container with elements.
40      @Test
41      @DisplayName("The removeAll method works on a container with elements")
42      default void testRemoveAllOnContainerWithElements() {
43          List<E> values = elementProvider().createUniqueInstances(DEFAULT_SIZE);
44          Collection<E> collection = provider().createInstance(values);
45          if (supportsMethod(CollectionMethods.RemoveAll)) {
46              // Remove first element
47              E first = values.getFirst();
48              boolean changed = collection.removeAll(Collections.singleton(first));
49              assertTrue(changed);
50              assertDoesNotContain(first, collection);
51  
52              // Remove first element
53              E last = values.getLast();
54              changed = collection.removeAll(Collections.singleton(last));
55              assertTrue(changed);
56              assertFalse(collection.contains(last));
57  
58              // Remove the remaining elements
59              changed = collection.removeAll(values.subList(1, values.size() - 1));
60              assertTrue(changed);
61              assertContainsNone(values, collection);
62          } else {
63              assertThrows(UnsupportedOperationException.class, () -> collection.remove(values.getFirst()));
64          }
65      }
66  
67      /// Tests that the `removeAll` method works with incompatible objects.
68      @Test
69      @DisplayName("The removeAll method works with null elements")
70      default void testRemoveAllOnNullElement() {
71          if (supportsMethod(CollectionMethods.RemoveAll) && permitNulls()) {
72              Collection<E> collection = provider().createInstanceWithUniqueElements();
73              Collection<E> argument = Collections.singleton(null);
74              if (permitNulls()) {
75                  assertFalse(collection.removeAll(argument));
76              } else {
77                  assertThrows(NullPointerException.class, () -> collection.removeAll(argument));
78              }
79          }
80      }
81  
82      /// Tests that the `removeAll` method works with incompatible objects.
83      @Test
84      @DisplayName("The removeAll method works with incompatible types")
85      default void testRemoveAllOnIncompatibleObject() {
86          if (supportsMethod(CollectionMethods.RemoveAll)) {
87              Collection<E> collection = provider().createInstanceWithUniqueElements();
88              assertFalse(collection.removeAll(Collections.singleton(new MatchNothing())));
89          }
90      }

93      /// Tests that the `removeAll` method throws on a null collection.
94      /// # Implementation Notes
95      /// This test checks that the `removeAll` function throws the correct exception if it is called on a null
96      /// value. IntelliJ will detect a problem when it is called with a null argument since the method declaration
97      /// has a NotNull annotation. Since this is what we are trying to test, the inspection is suppressed here.
98      ///
99      /// Any implementations that use the `NotNull` annotation for the collection parameter may throw an
100     /// `IllegalArgumentException` here, so either exception type is accepted.
101     @SuppressWarnings("DataFlowIssue")
102     @Test
103     @DisplayName("The removeAll method throws on a null collection")
104     default void testRemoveAllThrowsOnNullCollection() {
105         if (supportsMethod(CollectionMethods.RemoveAll)) {
106             Collection<E> collection = provider().emptyInstance();
107             Assertions.assertThrowsAny(List.of(NullPointerException.class, IllegalArgumentException.class),
108                     () -> collection.removeAll(null));
109         }
110     }
