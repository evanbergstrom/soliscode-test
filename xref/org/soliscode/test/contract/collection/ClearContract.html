5   import org.soliscode.test.contract.CollectionMethods;

8   import java.util.Collection;
9   
10  import static org.junit.jupiter.api.Assertions.assertThrows;
11  import static org.soliscode.test.assertions.collection.CollectionAssertions.assertIsEmpty;
12  
13  /// This interface tests if a collection class has implemented the `clear()` method
14  /// correctly.
15  ///
16  /// @author evanbergstrom
17  /// @param <E> The element type being tested.
18  /// @param <C> The collection type being tested.
19  /// @see Collection#clear
20  /// @since 1.0
21  public interface ClearContract<E, C extends Collection<E>> extends CollectionContractSupport<E, C> {
22  
23      /// Tests that the `clear()` method works for an empty collection.
24      @Test
25      @DisplayName("Test clear works for an empty collection")
26      default void testClearOnEmptyCollection() {
27          Collection<E> collection = provider().emptyInstance();
28          if (supportsMethod(CollectionMethods.Clear)) {
29              collection.clear();
30              assertIsEmpty(collection);
31          } else {
32              assertThrows(UnsupportedOperationException.class, collection::clear);
33          }
34      }
35  
36      /// Tests that the `clear()` method works for a collection with elements.
37      @Test
38      @DisplayName("Test clear works for a collection with elements")
39      default void testClearOnCollectionWithElements() {
40          Collection<E> collection = provider().createSingleton(elementProvider().createInstance());
41          if (supportsMethod(CollectionMethods.Clear)) {
42              collection.clear();
43              assertIsEmpty(collection);
44          } else {
45              assertThrows(UnsupportedOperationException.class, collection::clear);
46          }
47      }
48  }
