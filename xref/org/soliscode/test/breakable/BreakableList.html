11  /// A list that can be broken in well-defined ways in order to test collection utilities or testing
12  /// classes.
13  ///
14  /// # Breaks
15  /// In addition to the breaks for [BreakableSequencedCollection], the breaks found in [?] are also supported.
16  ///
17  /// Builder methods are provided to make declaring a broken list easier, for example:
18  /// ```java
19  ///     BreakableList<Integer> broken = Breakables.buildList(1, 2)
20  ///         .withBreak(List.ADD_ALL_AT_INDEX_DOES_NOT_ADD_ANY_ELEMENTS)
21  ///         .build();
22  /// ```
23  /// @author evanbergstrom
24  /// @param <E> The elements type for the `List`.
25  /// @see CollectionMethods
26  /// @since 1.0.0
27  public class BreakableList<E> extends BreakableSequencedCollection<E> implements List<E> {

215     // Implements the [addAll][List#addAll] method from the [List] interface. This method can be broken using the
216     // following collection breaks:
217     ///
218     /// | Break                    | Description                                     |
219     /// | ------------------------ | ----------------------------------------------- |
220     /// | ADD_ALL_AT_INDEX_DOES_NOT_ADD_ANY_ELEMENTS | The `addAll` method will not add any elements to the collection. |
221     /// | ADD_ALL_AT_INDEX_ADDS_TO_THE_END |  The `addAll` method adds to the end of the collection. |
222     /// | ADD_ALL_AT_INDEX_ALWAYS_RETURNS_TRUE | The `addAll` method always return a result of `true`, even if the element is not added. |
223     /// | ADD_ALL_AT_INDEX_ALWAYS_RETURNS_FALSE | The `addAll` method always returns `false` |
224     /// | ADD_ALL_AT_INDEX_ALWAYS_RETURNS_OPPOSITE_VALUE | The `addAll` method always returns `true` |
225     /// | ADD_ALL_AT_INDEX_THROWS_WRONG_EXCEPTION_ON_BAD_INDEX | The `addAll` method throws the wrong exception when the index is out of bounds. |
226     /// | ADD_ALL_AT_INDEX_THROWS_WRONG_EXCEPTION_IF_NOT_SUPPORTED | The `addAll` method throws the wrong exception if it is not supported. |
227     /// | ADD_ALL_AT_INDEX_THROWS_WRONG_EXCEPTION_WHEN_ARGUMENT_IS_NULL | The `addAll` method throws the wrong exception when the argument is null. |
228     ///
229     /// A collection that has any of these breaks can be constructed using the builder:
230     /// ```java
231     ///     Collection<Integer> collection = Breakables.buildList(1,2,3,4,5)
232     ///         .withBreak(ADD_ALL_AT_INDEX_ALWAYS_RETURNS_TRUE)
233     ///         .build();
234     /// ```
235     /// @param index index at which to insert the first element from the specified collection.
236     /// @param c collection containing elements to be added to this list
237     /// @return 'true' if this list changed as a result of the call
238     /// @throws UnsupportedOperationException if the 'addAll' operation is not supported by this list
239     /// @throws ClassCastException if the class of an element of the specified collection prevents it from being added
240     ///                            to this list
241     /// @throws NullPointerException if the specified collection contains one or more null elements and this list does
242     ///                              not permit null elements, or if the specified collection is null
243     /// @throws IllegalArgumentException if some property of an element of the specified collection prevents it from
244     ///                                  being added to this list
245     /// @throws IndexOutOfBoundsException if the index is out of range ('index < 0 || index > size()')
246     /// @see List#addAll(int, Collection)
247     @SuppressWarnings("ConstantValue")
248     @Override
249     public boolean addAll(int index, @NotNull Collection<? extends E> c) {
250         if (supportsMethod(CollectionMethods.AddAll)) {
251             if ((index < 0 || index >= list.size()) && hasBreak(ADD_ALL_AT_INDEX_THROWS_WRONG_EXCEPTION_ON_BAD_INDEX)) {
252                 throw new RuntimeException();
253             }
254 
255             if (c == null && hasBreak(ADD_ALL_AT_INDEX_THROWS_WRONG_EXCEPTION_WHEN_ARGUMENT_IS_NULL)) {
256                 throw new RuntimeException();
257             }
258 
259             boolean result;
260             if (hasBreak(ADD_ALL_AT_INDEX_ADDS_TO_THE_END)) {
261                 result = list.addAll(c);
262             } else if (!hasBreak(ADD_ALL_AT_INDEX_DOES_NOT_ADD_ANY_ELEMENTS)) {
263                 result = list.addAll(index, c);
264             } else {
265                 result = false;
266             }
267 
268             if (hasBreak(ADD_ALL_AT_INDEX_ALWAYS_RETURNS_TRUE)) {
269                 return true;
270             } else if (hasBreak(ADD_ALL_AT_INDEX_ALWAYS_RETURNS_FALSE)) {
271                 return false;
272             } else if (hasBreak(ADD_ALL_AT_INDEX_ALWAYS_RETURNS_OPPOSITE_VALUE)) {
273                 return !result;
274             } else {
275                 return result;
276             }
277         } else{
278             if (hasBreak(ADD_ALL_AT_INDEX_THROWS_WRONG_EXCEPTION_IF_NOT_SUPPORTED)) {
279                 throw new RuntimeException();
280             } else {
281                 throw new UnsupportedOperationException();
282             }
283         }
284     }

286     // Implements the [replaceAll][List#replaceAll] method from the [List] interface. This method can be broken using the
287     // following collection breaks:
288     ///
289     /// | Break                    | Description                                     |
290     /// | ------------------------ | ----------------------------------------------- |
291     /// | REPLACE_ALL_DOES_NOT_REPLACE_ELEMENTS | The `replaceAll` method does not replace any elements |
292     /// | REPLACE_ALL_SKIPS_FIRST_ELEMENT | The `replaceAll` method skips the first element |
293     /// | REPLACE_ALL_SKIPS_LAST_ELEMENT | The `replaceAll` method skips the last element |
294     ///
295     /// A collection that has any of these breaks can be constructed using the builder:
296     /// ```java
297     ///     Collection<Integer> collection = Breakables.buildList(1,2,3,4,5)
298     ///         .withBreak(REPLACE_ALL_DOES_NOT_REPLACE_ELEMENTS)
299     ///         .build();
300     /// ```
301     /// @throws UnsupportedOperationException if the `replaceAll` operation is not supported by this list
302     /// @throws NullPointerException if the specified operator is null or if the operator result is a null value and
303     ///         this list does not permit null elements.
304     /// @see List#replaceAll(UnaryOperator)
305     @Override
306     public void replaceAll(final UnaryOperator<E> operator) {
307         if (supportsMethod(CollectionMethods.ReplaceAll)) {
308             int start = 0;
309             int end = list.size();
310             if (hasBreak(REPLACE_ALL_SKIPS_FIRST_ELEMENT)) {
311                 start = 1;
312             } else if (hasBreak(REPLACE_ALL_SKIPS_LAST_ELEMENT)) {
313                 end = size() -1;
314             } else if (hasBreak(REPLACE_ALL_DOES_NOT_REPLACE_ELEMENTS)) {
315                 return;
316             }
317             for (int i = start; i < end; i++) {
318                 E e = operator.apply(list.get(i));
319                 if (e == null && !permitsNulls()) {
320                     throw new NullPointerException();
321                 }
322                 list.set(i, e);
323             }
324         } else{
325             throw new UnsupportedOperationException("Unsupported method: addAll");
326         }
327     }
