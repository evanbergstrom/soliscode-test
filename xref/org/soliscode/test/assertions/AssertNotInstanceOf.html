5   import static org.junit.jupiter.api.AssertionFailureBuilder.assertionFailure;

7   /// A collection of utility methods that support asserting that an object is not an instance of a class.
8   ///
9   /// @author evanbergstrom
10  /// @since 1.0.0
11  public class AssertNotInstanceOf {
12  
13      private AssertNotInstanceOf() {}
14  
15      static void assertNotInstanceOf(final Class<?> expectedType, final Object actual) {
16          assertNotInstanceOf(expectedType, actual, (Object) null);
17      }
18  
19      static void assertNotInstanceOf(final Class<?> expectedType, final Object actual, final String message) {
20          assertNotInstanceOf(expectedType, actual, (Object) message);
21      }
22  
23      static void assertNotInstanceOf(final Class<?> expectedType, final Object actual, final Supplier<String> messageSupplier) {
24          assertNotInstanceOf(expectedType, actual, (Object) messageSupplier);
25      }
26  
27      private static void assertNotInstanceOf(final Class<?> expectedType, final Object actual, final Object messageOrSupplier) {
28          if (expectedType.isInstance(actual)) {
29              assertionFailure()
30                      .message(messageOrSupplier)
31                      .reason(actual == null ? "Unexpected null value" : "Unexpected type")
32                      .expected(expectedType)
33                      .actual(actual == null ? null : actual.getClass()) //
34                      .buildAndThrow();
35          }
36      }
37  }
